from collections import deque

def bfs(r,c):
    global r2, c2, count
    queue = deque();

    dx = [-2, -2, 0, 0, 2, 2]
    dy = [-1, 1, -2, 2, -1, 1]

    queue.append((r,c,0))
    visited.append((r,c))

    while queue:
        v = queue.popleft()

        if v[0] == r2 and v[1] == c2:
            print(v[2])
            return

        for i in range(6):
            nr = v[0] + dx[i]
            nc = v[1] + dy[i]

            if 0 <= nr < n and 0 <= nc < n:
                if (nr, nc) not in visited:
                   queue.append((nr, nc, v[2]+1))
                   visited.append((nr, nc))
                   
    print(-1)

n = int(input())
