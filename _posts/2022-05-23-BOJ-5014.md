```python
import sys
from collections import deque

def bfs(v):
    dx = [u, -d]

    queue = deque()
    queue.append((v, 0))
    visited[v] = 1

    while queue:
        v = queue.popleft()

        if v[0] == g:
            print(v[1])
            return

        for i in range(2):
            nx = v[0] + dx[i]

            if 0 < nx <= f and visited[nx] == 0:
                visited[nx] = 1
                queue.append((nx, v[1]+1))

    print("use the stairs")           

f, s, g, u, d = map(int, sys.stdin.readline().split())
visited = [0] * (f+1)
bfs(s)
```
