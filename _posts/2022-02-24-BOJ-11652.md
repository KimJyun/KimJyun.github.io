```python
import sys

n = int(input())
cards = dict()

for _ in range(n):
    data = int(sys.stdin.readline())

    if data in cards:
        cards[data] += 1
    else:
        cards[data] = 1
    
cards = sorted(cards.items(), key=lambda x : ( -x[1], x[0] ))
print(cards[0][0])
```
